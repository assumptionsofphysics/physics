\begin{align*}
\overline{c_2} c_1 &= (a_2 - b_2 i) (a_1 + b_1 i) \\
&= a_1 a_2 + b_1 b_2 + \frac {a_1 b_2 - a_2 b_1}{i} \\
&= r_2 e^{-i\theta_2} r_1 e^{i\theta_1} \\
&= r_1 r_2 e^{-i\theta} \\
&= r_1 r_2 cos(\theta) + \frac {r_1 r_2 sin(\theta)}{i} \\
&= c_1 \cdot c_2 + \frac {c_1 \wedge c_2}{i} \\
\end{align*}
