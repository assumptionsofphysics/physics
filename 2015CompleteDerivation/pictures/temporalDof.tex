		\filldraw[black,fill=blue,opacity=0.5] (0,0) -- ($(p2)$) -- ($(p3)$) -- cycle;

		\coordinate (h7) at (\xvar + \avar,- \avar);
		\coordinate (h8) at (\xvar + \avar - \dvar,\avar + \dvar);

		% angle made by the coordinates to rotate the first dq dp square
		\newcommand{\phivar}{atan2(\dvar , \avar)};

		%drawing the shape on the right
		\filldraw[black,fill=green,opacity=0.5] ($(h0)$) -- ($(h2)$) -- ($(h3)$) -- cycle;

		\filldraw[black,fill=green,fill opacity=0.1,draw opacity=.2,text opacity =1,text=black,rotate around={0:($(h0)$)}] ($(h0)$) rectangle ($(h7)$) node[pos=.5] {\large $\omega$};

		\draw[black,fill=green,fill opacity=0.1,draw opacity=.2,text opacity =1,text=black,rotate around={0:($(h0)$)}] ($(h2)$) rectangle ($(h5)$) node[pos=.5] {\large $d\hat{t}\wedge d\hat{E}$};
